# °F/°C Toggle Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a click-to-toggle temperature unit control on the hero temperature, persist the choice, and propagate it to every temperature display in the dashboard.

**Architecture:** Pure conversion utilities in `shared/units.ts` (full TDD coverage). Unit preference lives in `App.tsx` state + `localStorage`, mirroring the existing dismissed-alerts persistence pattern, and is prop-drilled into the three display panels. Server continues to serve °F unchanged; conversion happens at the React render boundary.

**Tech Stack:** React 18, TypeScript, Vitest (existing). No new dependencies.

**Spec reference:** [docs/superpowers/specs/2026-04-17-v1.2-roadmap-design.md](../specs/2026-04-17-v1.2-roadmap-design.md) Section 1.

**Test scope note:** The codebase has no React/UI tests yet — only pure-logic tests in `shared/` and `server/`. This plan follows that established pattern: full TDD on `shared/units.ts` conversions, manual smoke validation for the React wiring. Adding React Testing Library + jsdom for one feature would conflict with the CLAUDE.md "minimize dependencies" hard rule.

```typescript
import type { Trend } from './types';

export type TempUnit = 'F' | 'C';

export function convertTempF(value: number, unit: TempUnit): number {
  if (unit === 'F') return value;
  return (value - 32) * 5 / 9;
}

export function scaleTempTrend(t: Trend, unit: TempUnit): Trend {
  if (unit === 'F') return t;
  return { ...t, deltaPerHour: t.deltaPerHour * 5 / 9 };
}
```

```typescript
// Temperature unit preference. Lives at App level alongside dismissed
// alerts so all child panels see one source of truth.
const UNITS_KEY = 'skyframe.units';

function loadUnits(): TempUnit {
  try {
    const raw = localStorage.getItem(UNITS_KEY);
    return raw === 'C' ? 'C' : 'F';
  } catch {
    return 'F';
  }
}

function saveUnits(unit: TempUnit): void {
  try {
    localStorage.setItem(UNITS_KEY, unit);
  } catch {
    // Quota exceeded or storage unavailable — silently degrade.
  }
}
```

```typescript
function computeChartPoints(
  hourly: HourlyPeriod[],
  units: TempUnit,
): {
  points: Array<{ x: number; y: number; temp: number }>;
  minTemp: number;
  maxTemp: number;
} {
  const temps = hourly.map((h) => convertTempF(h.tempF, units));
  const minTemp = Math.floor(Math.min(...temps) / 2) * 2;
  const maxTemp = Math.ceil(Math.max(...temps) / 2) * 2;
  const range = Math.max(1, maxTemp - minTemp);

  const columnWidth = SVG_WIDTH / hourly.length;
  const yScale = Y_BOTTOM - Y_TOP;

  const points = hourly.map((h, i) => {
    const t = convertTempF(h.tempF, units);
    return {
      x: (i + 0.5) * columnWidth,
      y: Y_TOP + ((maxTemp - t) / range) * yScale,
      temp: t,
    };
  });

  return { points, minTemp, maxTemp };
}
```

```typescript
export function OutlookPanel({ daily, units }: OutlookPanelProps) {
  if (daily.length === 0) return null;

  // Compute shared scale from all days' highs/lows in chosen unit
  const lows = daily.map((d) => convertTempF(d.lowF, units));
  const highs = daily.map((d) => convertTempF(d.highF, units));
  const scaleMin = Math.floor(Math.min(...lows) / 2) * 2;
  const scaleMax = Math.ceil(Math.max(...highs) / 2) * 2;
  const scaleRange = Math.max(1, scaleMax - scaleMin);

  const scalePoints = [
    scaleMin,
    Math.round(scaleMin + scaleRange * 0.25),
    Math.round(scaleMin + scaleRange * 0.5),
    Math.round(scaleMin + scaleRange * 0.75),
    scaleMax,
  ];

  return (
    <div className="hud-outlook-section">
      <div className="hud-section-label">
        <span>■ 7-DAY OUTLOOK &nbsp;·&nbsp; KMKE &nbsp;/&nbsp; MKX GRID 88,58 &nbsp;/&nbsp; WIZ066</span>
        <span>RANGE {scaleMin}° — {scaleMax}°{units}</span>
      </div>

      <div className="outlook-scale">
        <div></div><div></div><div></div>
        <div className="scale-axis">
          {scalePoints.map((n) => <span key={n}>{n}°</span>)}
        </div>
        <div></div>
      </div>

      <div className="outlook">
        {daily.map((day) => {
          const lo = convertTempF(day.lowF, units);
          const hi = convertTempF(day.highF, units);
          const leftPct = ((lo - scaleMin) / scaleRange) * 100;
          const rightPct = ((scaleMax - hi) / scaleRange) * 100;

          return (
            <OutlookRow
              key={day.dateISO}
              day={day}
              displayLow={Math.round(lo)}
              displayHigh={Math.round(hi)}
              leftPct={leftPct}
              rightPct={rightPct}
            />
          );
        })}
      </div>
    </div>
  );
}
```
